82  inline void loadTestNamesFromFile( ConfigData& config, std::string const& _filename ) {
83  std::ifstream f( _filename.c_str() );
84  if( !f.is_open() )
85  throw std::domain_error( "Unable to load input file: " + _filename );
86 
87  std::string line;
88  while( std::getline( f, line ) ) {
89  line = trim(line);
90  if( !line.empty() && !startsWith( line, '#' ) ) {
91  if( !startsWith( line, '"' ) )
92  line = '"' + line + '"';
93  addTestOrTags( config, line + ',' );
94  }
95  }
96  }
