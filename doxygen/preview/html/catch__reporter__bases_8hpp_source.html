169  virtual void sectionStarting( SectionInfo const& sectionInfo ) CATCH_OVERRIDE {
170  SectionStats incompleteStats( sectionInfo, Counts(), 0, false );
171  Ptr<SectionNode> node;
172  if( m_sectionStack.empty() ) {
173  if( !m_rootSection )
174  m_rootSection = new SectionNode( incompleteStats );
175  node = m_rootSection;
176  }
177  else {
178  SectionNode& parentNode = *m_sectionStack.back();
179  SectionNode::ChildSections::const_iterator it =
180  std::find_if( parentNode.childSections.begin(),
181  parentNode.childSections.end(),
182  BySectionInfo( sectionInfo ) );
183  if( it == parentNode.childSections.end() ) {
184  node = new SectionNode( incompleteStats );
185  parentNode.childSections.push_back( node );
186  }
187  else
188  node = *it;
189  }
190  m_sectionStack.push_back( node );
191  m_deepestSection = node;
192  }

196  virtual bool assertionEnded( AssertionStats const& assertionStats ) CATCH_OVERRIDE {
197  assert( !m_sectionStack.empty() );
198  SectionNode& sectionNode = *m_sectionStack.back();
199  sectionNode.assertions.push_back( assertionStats );
200  // AssertionResult holds a pointer to a temporary DecomposedExpression,
201  // which getExpandedExpression() calls to build the expression string.
202  // Our section stack copy of the assertionResult will likely outlive the
203  // temporary, so it must be expanded or discarded now to avoid calling
204  // a destroyed object later.
205  prepareExpandedExpression( sectionNode.assertions.back().assertionResult );
206  return true;
207  }
208  virtual void sectionEnded( SectionStats const& sectionStats ) CATCH_OVERRIDE {
209  assert( !m_sectionStack.empty() );
210  SectionNode& node = *m_sectionStack.back();
211  node.stats = sectionStats;
212  m_sectionStack.pop_back();
213  }

215  Ptr<TestCaseNode> node = new TestCaseNode( testCaseStats );
216  assert( m_sectionStack.size() == 0 );
