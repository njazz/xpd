56  Totals runTests( Ptr<Config> const& config ) {
57 
58  Ptr<IConfig const> iconfig = config.get();
59 
60  Ptr<IStreamingReporter> reporter = makeReporter( config );
61  reporter = addListeners( iconfig, reporter );
62 
63  RunContext context( iconfig, reporter );
64 
65  Totals totals;
66 
67  context.testGroupStarting( config->name(), 1, 1 );
68 
69  TestSpec testSpec = config->testSpec();
70  if( !testSpec.hasFilters() )
71  testSpec = TestSpecParser( ITagAliasRegistry::get() ).parse( "~[.]" ).testSpec(); // All not hidden tests
72 
73  std::vector<TestCase> const& allTestCases = getAllTestCasesSorted( *iconfig );
74  for( std::vector<TestCase>::const_iterator it = allTestCases.begin(), itEnd = allTestCases.end();
75  it != itEnd;
76  ++it ) {
77  if( !context.aborting() && matchTest( *it, testSpec, *iconfig ) )
78  totals += context.runTest( *it );
79  else
80  reporter->skipTest( *it );
81  }
82 
83  context.testGroupEnded( iconfig->name(), totals, 1, 1 );
84  return totals;
85  }

87  void applyFilenamesAsTags( IConfig const& config ) {
88  std::vector<TestCase> const& tests = getAllTestCasesSorted( config );
89  for(std::size_t i = 0; i < tests.size(); ++i ) {
90  TestCase& test = const_cast<TestCase&>( tests[i] );
91  std::set<std::string> tags = test.tags;
92 
93  std::string filename = test.lineInfo.file;
94  std::string::size_type lastSlash = filename.find_last_of( "\\/" );
95  if( lastSlash != std::string::npos )
96  filename = filename.substr( lastSlash+1 );
97 
98  std::string::size_type lastDot = filename.find_last_of( '.' );
99  if( lastDot != std::string::npos )
100  filename = filename.substr( 0, lastDot );
101 
102  tags.insert( '#' + filename );
103  setTags( test, tags );
104  }

134  int applyCommandLine( int argc, char const* const* const argv, OnUnusedOptions::DoWhat unusedOptionBehaviour = OnUnusedOptions::Fail ) {
