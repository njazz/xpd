87  Totals runTest( TestCase const& testCase ) {
88  Totals prevTotals = m_totals;
89 
90  std::string redirectedCout;
91  std::string redirectedCerr;
92 
93  TestCaseInfo testInfo = testCase.getTestCaseInfo();
94 
95  m_reporter->testCaseStarting( testInfo );
96 
97  m_activeTestCase = &testCase;
98 
99 
100  do {
101  ITracker& rootTracker = m_trackerContext.startRun();
102  assert( rootTracker.isSectionTracker() );
103  static_cast<SectionTracker&>( rootTracker ).addInitialFilters( m_config->getSectionsToRun() );
104  do {
105  m_trackerContext.startCycle();
106  m_testCaseTracker = &SectionTracker::acquire( m_trackerContext, TestCaseTracking::NameAndLocation( testInfo.name, testInfo.lineInfo ) );
107  runCurrentTest( redirectedCout, redirectedCerr );
108  }
109  while( !m_testCaseTracker->isSuccessfullyCompleted() && !aborting() );
110  }
111  // !TBD: deprecated - this will be replaced by indexed trackers
112  while( getCurrentContext().advanceGeneratorsForCurrentTest() && !aborting() );
113 
114  Totals deltaTotals = m_totals.delta( prevTotals );
115  if( testInfo.expectedToFail() && deltaTotals.testCases.passed > 0 ) {
116  deltaTotals.assertions.failed++;
117  deltaTotals.testCases.passed--;
118  deltaTotals.testCases.failed++;
119  }
120  m_totals.testCases += deltaTotals.testCases;
121  m_reporter->testCaseEnded( TestCaseStats( testInfo,
122  deltaTotals,
123  redirectedCout,
124  redirectedCerr,
125  aborting() ) );
126 
127  m_activeTestCase = CATCH_NULL;
128  m_testCaseTracker = CATCH_NULL;
129 
130  return deltaTotals;
131  }

250  // Don't rebuild the result -- the stringification itself can cause more fatal errors

254  tempResult.message = message;

257  getResultCapture().assertionEnded(result);

259  handleUnfinishedSections();
260 
261  // Recreate section for test case (as we will lose the one that was in scope)
262  TestCaseInfo const& testCaseInfo = m_activeTestCase->getTestCaseInfo();
263  SectionInfo testCaseSection( testCaseInfo.lineInfo, testCaseInfo.name, testCaseInfo.description );
264 
265  Counts assertions;
266  assertions.failed = 1;
267  SectionStats testCaseSectionStats( testCaseSection, assertions, 0, false );
268  m_reporter->sectionEnded( testCaseSectionStats );
269 
270  TestCaseInfo testInfo = m_activeTestCase->getTestCaseInfo();
271 
272  Totals deltaTotals;
273  deltaTotals.testCases.failed = 1;
274  m_reporter->testCaseEnded( TestCaseStats( testInfo,
275  deltaTotals,
276  std::string(),
277  std::string(),
278  false ) );
279  m_totals.testCases.failed++;
280  testGroupEnded( std::string(), m_totals, 1, 1 );

292  void runCurrentTest( std::string& redirectedCout, std::string& redirectedCerr ) {
293  TestCaseInfo const& testCaseInfo = m_activeTestCase->getTestCaseInfo();
294  SectionInfo testCaseSection( testCaseInfo.lineInfo, testCaseInfo.name, testCaseInfo.description );
295  m_reporter->sectionStarting( testCaseSection );
296  Counts prevAssertions = m_totals.assertions;
297  double duration = 0;
298  m_shouldReportUnexpected = true;
299  try {
300  m_lastAssertionInfo = AssertionInfo( "TEST_CASE", testCaseInfo.lineInfo, "", ResultDisposition::Normal );
301 
302  seedRng( *m_config );
303 
304  Timer timer;
305  timer.start();
306  if( m_reporter->getPreferences().shouldRedirectStdOut ) {
307  StreamRedirect coutRedir( Catch::cout(), redirectedCout );
308  StreamRedirect cerrRedir( Catch::cerr(), redirectedCerr );
309  invokeActiveTestCase();
310  }
311  else {
312  invokeActiveTestCase();
313  }
314  duration = timer.getElapsedSeconds();
315  }
316  catch( TestFailureException& ) {
317  // This just means the test was aborted due to failure
318  }
319  catch(...) {
320  // Under CATCH_CONFIG_FAST_COMPILE, unexpected exceptions under REQUIRE assertions
321  // are reported without translation at the point of origin.
322  if (m_shouldReportUnexpected) {
323  makeUnexpectedResultBuilder().useActiveException();
324  }
325  }
326  m_testCaseTracker->close();
327  handleUnfinishedSections();
328  m_messages.clear();
329 
330  Counts assertions = m_totals.assertions - prevAssertions;
331  bool missingAssertions = testForMissingAssertions( assertions );
332 
333  if( testCaseInfo.okToFail() ) {
334  std::swap( assertions.failedButOk, assertions.failed );
335  m_totals.assertions.failed -= assertions.failedButOk;
336  m_totals.assertions.failedButOk += assertions.failedButOk;
337  }
338 
339  SectionStats testCaseSectionStats( testCaseSection, assertions, duration, missingAssertions );
340  m_reporter->sectionEnded( testCaseSectionStats );
341  }

358  void handleUnfinishedSections() {
359  // If sections ended prematurely due to an exception we stored their
360  // infos here so we can tear them down outside the unwind process.
361  for( std::vector<SectionEndInfo>::const_reverse_iterator it = m_unfinishedSections.rbegin(),
362  itEnd = m_unfinishedSections.rend();
363  it != itEnd;

365  sectionEnded( *it );
366  m_unfinishedSections.clear();
367  }
