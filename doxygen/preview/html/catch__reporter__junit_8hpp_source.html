22  std::string getCurrentTimestamp() {
23  // Beware, this is not reentrant because of backward compatibility issues
24  // Also, UTC only, again because of backward compatibility (%z is C++11)
25  time_t rawtime;
26  std::time(&rawtime);
27  const size_t timeStampSize = sizeof("2017-01-16T17:06:45Z");
28 
29 #ifdef _MSC_VER
30  std::tm timeInfo = {};
31  gmtime_s(&timeInfo, &rawtime);
32 #else
33  std::tm* timeInfo;
34  timeInfo = std::gmtime(&rawtime);
35 #endif
36 
37  char timeStamp[timeStampSize];
38  const char * const fmt = "%Y-%m-%dT%H:%M:%SZ";
39 
40 #ifdef _MSC_VER
41  std::strftime(timeStamp, timeStampSize, fmt, &timeInfo);
42 #else
43  std::strftime(timeStamp, timeStampSize, fmt, timeInfo);
44 #endif
45  return std::string(timeStamp);
46  }

107  void writeGroup( TestGroupNode const& groupNode, double suiteTime ) {
108  XmlWriter::ScopedElement e = xml.scopedElement( "testsuite" );
109  TestGroupStats const& stats = groupNode.value;
110  xml.writeAttribute( "name", stats.groupInfo.name );
111  xml.writeAttribute( "errors", unexpectedExceptions );
112  xml.writeAttribute( "failures", stats.totals.assertions.failed-unexpectedExceptions );
113  xml.writeAttribute( "tests", stats.totals.assertions.total() );
114  xml.writeAttribute( "hostname", "tbd" ); // !TBD
115  if( m_config->showDurations() == ShowDurations::Never )
116  xml.writeAttribute( "time", "" );
117  else
118  xml.writeAttribute( "time", suiteTime );
119  xml.writeAttribute( "timestamp", getCurrentTimestamp() );
120 
121  // Write test cases
122  for( TestGroupNode::ChildNodes::const_iterator
123  it = groupNode.children.begin(), itEnd = groupNode.children.end();
124  it != itEnd;
125  ++it )
126  writeTestCase( **it );
127 
128  xml.scopedElement( "system-out" ).writeText( trim( stdOutForSuite.str() ), false );
129  xml.scopedElement( "system-err" ).writeText( trim( stdErrForSuite.str() ), false );
130  }

132  void writeTestCase( TestCaseNode const& testCaseNode ) {
133  TestCaseStats const& stats = testCaseNode.value;
134 
135  // All test cases have exactly one section - which represents the
136  // test case itself. That section may have 0-n nested sections
137  assert( testCaseNode.children.size() == 1 );
138  SectionNode const& rootSection = *testCaseNode.children.front();
139 
140  std::string className = stats.testInfo.className;
141 
142  if( className.empty() ) {
143  if( rootSection.childSections.empty() )
144  className = "global";
145  }
146  writeSection( className, "", rootSection );
147  }

149  void writeSection( std::string const& className,
150  std::string const& rootName,
151  SectionNode const& sectionNode ) {
152  std::string name = trim( sectionNode.stats.sectionInfo.name );
153  if( !rootName.empty() )
154  name = rootName + '/' + name;
155 
156  if( !sectionNode.assertions.empty() ||
157  !sectionNode.stdOut.empty() ||
158  !sectionNode.stdErr.empty() ) {
159  XmlWriter::ScopedElement e = xml.scopedElement( "testcase" );
160  if( className.empty() ) {
161  xml.writeAttribute( "classname", name );
162  xml.writeAttribute( "name", "root" );
163  }
164  else {
165  xml.writeAttribute( "classname", className );
166  xml.writeAttribute( "name", name );
167  }
168  xml.writeAttribute( "time", Catch::toString( sectionNode.stats.durationInSeconds ) );
169 
170  writeAssertions( sectionNode );
171 
172  if( !sectionNode.stdOut.empty() )
173  xml.scopedElement( "system-out" ).writeText( trim( sectionNode.stdOut ), false );
174  if( !sectionNode.stdErr.empty() )
175  xml.scopedElement( "system-err" ).writeText( trim( sectionNode.stdErr ), false );
176  }
177  for( SectionNode::ChildSections::const_iterator
178  it = sectionNode.childSections.begin(),
179  itEnd = sectionNode.childSections.end();
180  it != itEnd;
181  ++it )
182  if( className.empty() )
183  writeSection( name, "", **it );
184  else
185  writeSection( className, name, **it );
186  }

188  void writeAssertions( SectionNode const& sectionNode ) {
189  for( SectionNode::Assertions::const_iterator

195  void writeAssertion( AssertionStats const& stats ) {
196  AssertionResult const& result = stats.assertionResult;
197  if( !result.isOk() ) {
198  std::string elementName;
199  switch( result.getResultType() ) {
200  case ResultWas::ThrewException:
201  case ResultWas::FatalErrorCondition:
202  elementName = "error";
203  break;
204  case ResultWas::ExplicitFailure:
205  elementName = "failure";
206  break;
207  case ResultWas::ExpressionFailed:
208  elementName = "failure";
209  break;
210  case ResultWas::DidntThrowException:
211  elementName = "failure";
212  break;
213 
214  // We should never see these here:
215  case ResultWas::Info:
216  case ResultWas::Warning:
217  case ResultWas::Ok:
218  case ResultWas::Unknown:
219  case ResultWas::FailureBit:
220  case ResultWas::Exception:
221  elementName = "internalError";
222  break;
223  }
224 
225  XmlWriter::ScopedElement e = xml.scopedElement( elementName );
226 
227  xml.writeAttribute( "message", result.getExpandedExpression() );
228  xml.writeAttribute( "type", result.getTestMacroName() );
229 
230  std::ostringstream oss;
231  if( !result.getMessage().empty() )
232  oss << result.getMessage() << '\n';
233  for( std::vector<MessageInfo>::const_iterator
234  it = stats.infoMessages.begin(),
235  itEnd = stats.infoMessages.end();
236  it != itEnd;
237  ++it )
238  if( it->type == ResultWas::Info )
239  oss << it->message << '\n';
240 
241  oss << "at " << result.getSourceInfo();
242  xml.writeText( oss.str(), false );
243  }
244  }
